CREATE DATABASE BIBLIOTECA;
USE BIBLIOTECA;

CREATE TABLE LIVRO (
   ID INT NOT NULL AUTO_INCREMENT,
   TITULO VARCHAR(50),
   AUTOR VARCHAR(50),
   ANO INT,
   PRIMARY KEY(ID)
);
